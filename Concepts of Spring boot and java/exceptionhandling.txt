import lombok.extern.slf4j.Slf4j;
import org.springframework.dao.DataIntegrityViolationException;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.FieldError;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;
import org.springframework.web.context.request.WebRequest;

import java.nio.file.AccessDeniedException;
import java.time.LocalDateTime;
import java.util.HashMap;
import java.util.Map;

@RestControllerAdvice
@Slf4j
public class RealTimeGlobalExceptionHandler {

    // 1. ALL CUSTOM BUSINESS EXCEPTIONS (Clean & Centralized)
    @ExceptionHandler(BaseBusinessException.class)
    public ResponseEntity<ApiErrorResponse> handleBusinessExceptions(BaseBusinessException ex, WebRequest request) {
        log.warn("Business rule violation: {} - Code: {}", ex.getMessage(), ex.getErrorCode());
        
        ApiErrorResponse response = buildErrorResponse(ex.getHttpStatus(), ex.getMessage(), ex.getErrorCode(), request);
        return new ResponseEntity<>(response, ex.getHttpStatus());
    }

    // 2. SPRING VALIDATION EXCEPTIONS (When DTO validation e.g., @NotBlank fails)
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<ApiErrorResponse> handleValidationExceptions(MethodArgumentNotValidException ex, WebRequest request) {
        log.warn("Validation failed for request: {}", request.getDescription(false));
        
        Map<String, String> errors = new HashMap<>();
        ex.getBindingResult().getAllErrors().forEach((error) -> {
            String fieldName = ((FieldError) error).getField();
            String errorMessage = error.getDefaultMessage();
            errors.put(fieldName, errorMessage);
        });

        ApiErrorResponse response = ApiErrorResponse.builder()
                .timestamp(LocalDateTime.now())
                .status(HttpStatus.BAD_REQUEST.value())
                .error(HttpStatus.BAD_REQUEST.getReasonPhrase())
                .customErrorCode("VALIDATION_ERROR")
                .message("Input payload validation failed")
                .validationErrors(errors)
                .path(request.getDescription(false).replace("uri=", ""))
                .build();

        return new ResponseEntity<>(response, HttpStatus.BAD_REQUEST);
    }

    // 3. DATABASE CONSTRAINT VIOLATIONS (Catching unique constraint drops before they leak)
    @ExceptionHandler(DataIntegrityViolationException.class)
    public ResponseEntity<ApiErrorResponse> handleDatabaseConstraints(DataIntegrityViolationException ex, WebRequest request) {
        log.error("Database constraint violation detected!", ex);
        
        ApiErrorResponse response = buildErrorResponse(
                HttpStatus.CONFLICT, 
                "Database error: Data integrity violation or duplicate key.", 
                "DB_CONSTRAINT_ERROR", 
                request
        );
        return new ResponseEntity<>(response, HttpStatus.CONFLICT);
    }

    // 4. SECURITY / AUTHORIZATION EXCEPTIONS
    @ExceptionHandler(AccessDeniedException.class)
    public ResponseEntity<ApiErrorResponse> handleAccessDenied(AccessDeniedException ex, WebRequest request) {
        log.warn("Unauthorized access attempt on path: {}", request.getDescription(false));
        
        ApiErrorResponse response = buildErrorResponse(
                HttpStatus.FORBIDDEN, 
                "You do not have permission to access this resource.", 
                "AUTH_FORBIDDEN", 
                request
        );
        return new ResponseEntity<>(response, HttpStatus.FORBIDDEN);
    }

    // 5. THE ULTIMATE CATCH-ALL (Prevents raw stack traces from exposing system internals)
    @ExceptionHandler(Exception.class)
    public ResponseEntity<ApiErrorResponse> handleAllUncaughtExceptions(Exception ex, WebRequest request) {
        // Critical: Log the full stack trace internally so developers can fix it via logs
        log.error("CRITICAL UNCAUGHT EXCEPTION: ", ex);
        
        ApiErrorResponse response = buildErrorResponse(
                HttpStatus.INTERNAL_SERVER_ERROR, 
                "A critical internal server error occurred. Please contact support.", 
                "SYS_INTERNAL_ERROR", 
                request
        );
        return new ResponseEntity<>(response, HttpStatus.INTERNAL_SERVER_ERROR);
    }

    // Helper method to keep code DRY
    private ApiErrorResponse buildErrorResponse(HttpStatus status, String message, String customCode, WebRequest request) {
        return ApiErrorResponse.builder()
                .timestamp(LocalDateTime.now())
                .status(status.value())
                .error(status.getReasonPhrase())
                .customErrorCode(customCode)
                .message(message)
                .path(request.getDescription(false).replace("uri=", ""))
                .build();
    }
}

What is the difference between @ControllerAdvice and @RestControllerAdvice?
What they are testing: Your understanding of Spring MVC basics.

The Answer: @RestControllerAdvice is a convenience annotation that combines @ControllerAdvice and @ResponseBody.

With @ControllerAdvice, methods return a view name (HTML page) by default unless you add @ResponseBody to every method.

With @RestControllerAdvice, Spring automatically serializes the returned object (like your ErrorResponse) directly into JSON/XML in the HTTP response body.



How do you handle exceptions that occur before reaching the Controller (e.g., in a Filter or Spring Security)?
What they are testing: Understanding the Spring request lifecycle.

The Answer: @RestControllerAdvice only catches exceptions thrown within the DispatcherServlet's scope (Controller, Service, Repository layers). It cannot catch exceptions thrown inside Servlet Filters or Spring Security filters because filters execute before the request ever reaches the dispatcher. To handle filter exceptions, you must either handle them inside the filter directly or configure an AuthenticationEntryPoint / AccessDeniedHandler in your Spring Security configuration.


If two @ExceptionHandler methods match the same exception hierarchy (e.g., one catches IOException and another catches Exception), which one will run?
What they are testing: Spring's exception matching algorithm.

The Answer: Spring always looks for the most specific match first. If an IOException is thrown, the handler explicitly annotated for IOException will execute. The generic Exception handler will only pick it up if a more specific handler doesn't exist.


How do you maintain the order of execution if you have multiple @RestControllerAdvice classes?
What they are testing: Microservices or modular architecture design.

The Answer: If you have multiple advice classes (e.g., one in a shared library module and one in your main application), you can control their priority using the @Order annotation or by implementing the Ordered interface. Lower values have higher priority (e.g., @Order(1) runs before @Order(2)).


f a controller method has a try-catch block, will @RestControllerAdvice still catch that exception?
What they are testing: Your understanding of Java basics and Spring's interceptor lifecycle.
The Answer: No. Spring's AOP (Aspect-Oriented Programming) proxy intercepts exceptions after they escape the controller layer. If you catch and handle the exception inside the controller method, it never bubbles up to Spring, so the global exception handler will never see it.

Java calls alex2.hashCode(): It gets the number 101. Java instantly jumps straight to the "Bucket 101" in memory. It doesn't look anywhere else.

Java looks inside Bucket 101: It sees alex1 is already sitting there.

Java calls alex1.equals(alex2): It checks if the names and IDs match perfectly. They do!

Database Patterns (How to manage data)
Data is arguably the hardest part of microservices. Sharing a database creates tight coupling, but separating them creates consistency issues.

Database per Service: The golden rule of microservices. Each service owns its database, and other services can only access that data through the owning service's API. This ensures loose coupling.

Saga Pattern: Because you can't easily use traditional ACID transactions across multiple databases, a Saga manages distributed transactions as a sequence of local transactions. If one step fails, the Saga executes compensating transactions to undo the preceding steps.


Strangler Fig Pattern: Used for migrating a legacy monolith to microservices. You incrementally extract functionalities into new microservices and route traffic to them, gradually "strangling" the old monolith until it can be retired.



Integration & Routing Patterns (How they communicate)
Clients (web, mobile) need a cohesive way to talk to dozens of fragmented backend services.

API Gateway: Acts as a single entry point into the system for all clients. It handles request routing, composition, and cross-cutting concerns like authentication, rate limiting, and SSL termination.

BFF (Backend for Frontend): A variation of the API Gateway where you create a separate API Gateway for each type of UI (e.g., one for the mobile app, one for the web app). This prevents a single gateway from becoming a bottleneck and allows each client to get exactly the data shape it needs.









502 Bad Gateway error



The Upstream Service is Down or Unreachable
This is the most common cause. The gateway is trying to route traffic to a service that is not there.

Application Crash: The Spring Boot application has crashed (e.g., due to an OutOfMemoryError or fatal exception) and the process has terminated.

Deployment Downtime: During a CI/CD pipeline deployment, if zero-downtime deployment (like blue-green or rolling updates) is not configured correctly, requests hitting the gateway while the new instance is starting up will result in a 502.

Failed Health Checks: The container orchestration system (like Kubernetes) or load balancer marks the Spring Boot pod/instance as unhealthy and terminates it, but the gateway hasn't updated its routing table yet.

2. Service Discovery (Registry) Sync Issues
If you are using a service registry like Netflix Eureka, HashiCorp Consul, or Kubernetes CoreDNS, the gateway relies on this registry to know where services live.

Stale Registry Data: When a Spring Boot instance scales down or crashes, it takes a few seconds (or minutes, depending on your cache configurations) for the registry to remove it. During this window, the API Gateway still routes traffic to the dead IP address.

Missing Instances: The service is running, but it failed to register itself with the discovery server due to misconfiguration (e.g., wrong eureka.client.serviceUrl.defaultZone).

3. Resource Exhaustion (Tomcat/Undertow)
Even if the Spring Boot application is technically "running," it might be in a state where it cannot accept or process new TCP connections.

Thread Pool Exhaustion: Spring Boot’s embedded server (Tomcat by default) has a maximum number of worker threads (default server.tomcat.threads.max=200). If all threads are busy (often due to blocked database queries or slow external API calls), the server will reject new connections, causing the gateway to throw a 502.

Connection Queue Full: If the thread pool is full, Tomcat places requests in a queue (the accept-count, default 100). If this queue also fills up, subsequent connections are immediately refused by the OS.



Firewall/Security Groups:


Check the Gateway Logs First: Find the exact IP and port the gateway was trying to reach when the 502 occurred.

Verify the IP: Check your Service Registry (Eureka/Consul) or Kubernetes Endpoints. Does that IP actually exist right now?

Check Upstream Logs: Look at the logs of the specific Spring Boot instance. If there are no logs for the request, the request never reached it (pointing to Network/Registry issues). If there are OutOfMemory or database timeout logs, you have a resource exhaustion issue.
