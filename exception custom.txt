import java.io.BufferedReader;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.io.PrintWriter; // Used with FileWriter for convenience
import java.util.Scanner;

// 1. Custom Exception Class
// This exception will be thrown when a specific custom condition is met.
class InvalidInputException extends Exception {
    // Constructor that takes a message
    public InvalidInputException(String message) {
        super(message);
    }

    // Constructor that takes a message and a cause (another Throwable)
    public InvalidInputException(String message, Throwable cause) {
        super(message, cause);
    }
}

public class ExceptionHandlingDemo {

    public static void main(String[] args) {
        System.out.println("--- Demonstrating Exception Handling in Java ---");

        // --- 1. Demonstrating ArithmeticException ---
        // This exception occurs when an exceptional arithmetic condition occurs,
        // such as division by zero.
        System.out.println("\n--- ArithmeticException Example ---");
        int numerator = 10;
        int denominator = 0; // Intentional division by zero
        try {
            System.out.println("Attempting division: " + numerator + " / " + denominator);
            int result = numerator / denominator; // This line will throw ArithmeticException
            System.out.println("Result of division: " + result); // This line will not be reached
        } catch (ArithmeticException e) {
            // Catch block for ArithmeticException
            System.err.println("Caught ArithmeticException: " + e.getMessage());
            System.err.println("Cause: Cannot divide by zero.");
        } finally {
            System.out.println("Arithmetic operation attempt finished (finally block).\n");
        }


        // --- 2. Demonstrating NullPointerException ---
        // This exception occurs when an application attempts to use null in a case
        // where an object is required, such as invoking a method or accessing a field
        // of a null object.
        System.out.println("--- NullPointerException Example ---");
        String text = null; // Intentionally set to null
        try {
            System.out.println("Attempting to get length of a null string...");
            // This line will throw NullPointerException because 'text' is null
            int length = text.length();
            System.out.println("Length of string: " + length); // This line will not be reached
        } catch (NullPointerException e) {
            // Catch block for NullPointerException
            System.err.println("Caught NullPointerException: " + e.getMessage());
            System.err.println("Cause: Attempted to access a method on a null object.");
        } finally {
            System.out.println("Null pointer operation attempt finished (finally block).\n");
        }


        // --- 3. Demonstrating try-with-resources ---
        // This feature ensures that each resource (an object that must be closed after
        // the program is finished with it) is closed at the end of the statement,
        // regardless of whether the try block completes normally or abruptly.
        System.out.println("--- Try-with-Resources Example ---");
        String fileName = "example.txt";

        // First, write some content to the file using try-with-resources
        System.out.println("Writing to file using try-with-resources...");
        try (FileWriter fileWriter = new FileWriter(fileName); // Resource 1
             PrintWriter printWriter = new PrintWriter(fileWriter)) { // Resource 2
            printWriter.println("Hello from try-with-resources!");
            printWriter.println("This line is also written.");
            System.out.println("Content successfully written to " + fileName);
        } catch (IOException e) {
            System.err.println("Caught IOException during file write: " + e.getMessage());
        }
        System.out.println("FileWriter and PrintWriter are automatically closed.\n");

        // Now, read content from the file using try-with-resources
        System.out.println("Reading from file using try-with-resources...");
        try (BufferedReader reader = new BufferedReader(new FileReader(fileName))) {
            String line;
            System.out.println("Content of " + fileName + ":");
            while ((line = reader.readLine()) != null) {
                System.out.println(line);
            }
            System.out.println("Content successfully read from " + fileName);
        } catch (IOException e) {
            System.err.println("Caught IOException during file read: " + e.getMessage());
        }
        System.out.println("BufferedReader and FileReader are automatically closed.\n");


        // --- 4. Demonstrating Custom Exception ---
        System.out.println("--- Custom Exception Example ---");
        Scanner scanner = new Scanner(System.in);
        try {
            System.out.print("Enter a number (must be positive and even): ");
            int number = scanner.nextInt();

            // Custom logic to throw the custom exception
            if (number <= 0) {
                throw new InvalidInputException("Number must be positive.");
            }
            if (number % 2 != 0) {
                throw new InvalidInputException("Number must be even.");
            }

            System.out.println("Valid input received: " + number);

        } catch (InvalidInputException e) {
            // Catch block for our custom exception
            System.err.println("Caught Custom Exception: " + e.getMessage());
            System.err.println("Please provide valid input as specified.");
        } catch (java.util.InputMismatchException e) {
            // Catch block for non-integer input
            System.err.println("Caught InputMismatchException: Invalid input. Please enter an integer.");
        } finally {
            System.out.println("Custom exception demonstration finished.\n");
            if (scanner != null) {
                scanner.close(); // Close the scanner to prevent resource leak
            }
        }

        System.out.println("--- End of Demonstration ---");
    }
}
