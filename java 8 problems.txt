import java.util.*;

public class Duplicates {

    public static Set<Integer> findDuplicates(int[] a) {
        Set<Integer> seen = new HashSet<>();
        Set<Integer> dup  = new LinkedHashSet<>();

        for (int num : a) {
            if (!seen.contains(num)) {   // if NOT seen → add to seen
                seen.add(num);
            } else {                     // already seen → duplicate
                dup.add(num);
            }
        }
        return dup;
    }

    public static void main(String[] args) {
        int[] a = {2, 3, 5, 7, 2, 9, 6, 3};
        System.out.println(findDuplicates(a));  // [2, 3]
    }
}

        int[] a = {2, 3, 5, 7, 2, 9, 6, 3};

        int left  = 0;
        int right = a.length - 1;

        while (left < right) {
            if (a[left] % 2 == 0) {
                left++;                    // even → already in place → move left
            } else if (a[right] % 2 != 0) {
                right--;                   // odd on right → already in place → move right
            } else {
                int temp    = a[left];     // swap odd(left) with even(right)
                a[left]     = a[right];
                a[right]    = temp;
                left++;
                right--;
            }
        }

        System.out.println(Arrays.toString(a));
    }

import java.util.Stack;

public class Main {
    public static void main(String[] args) {

        String s = "{[()]}"; 
        Stack<Character> stack = new Stack<>();
        boolean isValid = true;

        for (char c : s.toCharArray()) {
            // 1. If it's an opening bracket, push it onto the stack
            if (c == '(' || c == '{' || c == '[') {
                stack.push(c);
            } 
            // 2. If it's a closing bracket, we must validate it
            else if (c == ')' || c == '}' || c == ']') {
                
                // If stack is empty but we have a closing bracket, it's invalid (e.g., " ] ")
                if (stack.isEmpty()) {
                    isValid = false;
                    break;
                }

                // Pop the top element and check if it matches the closing bracket
                char top = stack.pop();
                if ((c == ')' && top != '(') ||
                    (c == '}' && top != '{') ||
                    (c == ']' && top != '[')) {
                    isValid = false; // Mismatch found!
                    break; 
                }
            }
        }

        // The string is only valid if we found no mismatches AND the stack is completely empty
        if (isValid && stack.isEmpty()) {
            System.out.println("Valid");
        } else {
            System.out.println("Invalid");
        }
    }
}

public class VeryEasyInsertArray {

    public static void main(String[] args) {

        // 1. Define an existing array
        int[] originalArray = {10, 20, 30, 40, 50};
        System.out.println("Original Array: " + Arrays.toString(originalArray));

        // 2. Define the element to insert and its position
        int elementToInsert = 25;
        int position = 2; // Insert at index 2 (between 20 and 30)

        // Basic validation (optional but good practice)
        if (position < 0 || position > originalArray.length) {
            System.out.println("Error: Invalid position for insertion.");
            return;
        }

        // 3. Create a new array that is one element larger
        int[] newArray = new int[originalArray.length + 1];

        // 4. Copy elements and insert the new element
        for (int i = 0; i < newArray.length; i++) {
            if (i < position) {
                // Copy elements before the insertion point directly
                newArray[i] = originalArray[i];
            } else if (i == position) {
                // Insert the new element at the specified position
                newArray[i] = elementToInsert;
            } else {
                // Copy elements from the original array, shifted one position to the right
                newArray[i] = originalArray[i - 1];
            }
        }

        // 5. Print the new array
        System.out.println("Element to insert: " + elementToInsert);
        System.out.println("Position to insert at: " + position);
        System.out.println("Array after insertion: " + Arrays.toString(newArray));
    }
}













public class ArrayMergeExample {
    public static void main(String[] args) {
        int[] arr1 = {1, 7};
        int[] arr2 = {2, 9, 11};
        int[] arr3 = new int[arr1.length + arr2.length];

        // --- 1. Copy arr1 into the beginning of arr3 ---
        for (int i = 0; i < arr1.length; i++) {
            arr3[i] = arr1[i];
        }
        
        // --- 2. Copy arr2 into arr3, starting AFTER arr1's elements ---
        // The index for arr3 must start where arr1 left off (at arr1.length).
        for (int i = 0; i < arr2.length; i++) {
            // arr3 index: i + arr1.length
            // arr2 index: i
            arr3[i + arr1.length] = arr2[i]; 
        }

        // --- 3. Print the merged array arr3 ---
        System.out.println("Merged Array (arr3):");
        for (int i : arr3) {
            System.out.println(i);
        }

        // --- 4. Fix the final loop logic (to find the last value in the first half) ---
        int d = 0;
        
        // The loop should iterate up to (but not including) arr3.length / 2
        // If arr3.length is 5, loop is for i=0 and i=1. The last element assigned is arr3[1].
        for (int i = 0; i < arr3.length / 2; i++) {
            d = arr3[i]; 
        }
        
        // This prints the value of the last element assigned to 'd' in the loop.
        // arr3 = {1, 7, 2, 9, 11}. arr3.length/2 = 2.
        // i=0: d = arr3[0] = 1
        // i=1: d = arr3[1] = 7
        // After loop, d = 7.
        System.out.println("\nLast element in the first half of arr3 (value of d): " + d);
    }

public class PalindromeChecker {

    /**
     * Checks if an integer is a palindrome.
     * A number is a palindrome if it reads the same backward as forward.
     *
     * @param originalNumber The integer to check.
     * @return true if the number is a palindrome, false otherwise.
     */
    public static boolean isPalindrome(int originalNumber) {
        // Handle negative numbers (they are generally not considered palindromes)
        if (originalNumber < 0) {
            return false;
        }

        int number = originalNumber; // Use a working copy of the number
        int reversedNumber = 0;

        while (number != 0) {
            // 1. Get the last digit
            int lastDigit = number % 10; 
            
            // 2. Add the last digit to the reversed number
            //    Multiply by 10 to shift existing digits to the left
            reversedNumber = reversedNumber * 10 + lastDigit;
            
            // 3. Remove the last digit from the original number
            number /= 10; 
        }

        // The original number is a palindrome if it equals the reversed number
        return originalNumber == reversedNumber;
    }

    public static void main(String[] args) {
        int numberToCheck = 82759825; // int numner = 82759825; from your query

        if (isPalindrome(numberToCheck)) {
            System.out.println(numberToCheck + " is a palindrome.");
        } else {
            System.out.println(numberToCheck + " is NOT a palindrome.");
        }
        
        // Example of a palindrome
        int palindromeExample = 12321;
        if (isPalindrome(palindromeExample)) {
            System.out.println(palindromeExample + " is a palindrome.");
        }
    }
}

class Solution {
    /**
     * Checks if a string is a palindrome, ignoring case, spaces, and punctuation.
     * * @param s The input string.
     * @return true if the string is a palindrome, false otherwise.
     */
    public boolean solution(String s) {
        // Step 1: Preprocess the string
        // Use a StringBuilder to efficiently build the cleaned string
        StringBuilder cleaned = new StringBuilder();
        
        // Iterate through each character of the input string
        for (char ch : s.toCharArray()) {
            // Check if the character is a letter or a digit
            if (Character.isLetterOrDigit(ch)) {
                // Convert to lowercase and append to the StringBuilder
                cleaned.append(Character.toLowerCase(ch));
            }
        }
        
        // Convert the StringBuilder to a String for comparison
        String cleanedString = cleaned.toString();
        
        // Step 2: Check for a palindrome using the two-pointer method
        int left = 0;
        int right = cleanedString.length() - 1;
        
        while (left < right) {
            if (cleanedString.charAt(left) != cleanedString.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }
        
        // If the loop completes, the string is a palindrome
        return true;
    }
}

Problem: Given the root of a binary tree, return the level order traversal of its nodes. This means visiting nodes level by level, from left to right.

import java.util.*;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;
    TreeNode(int x) { val = x; }
}

public class LevelOrderTraversal {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        if (root == null) {
            return result;
        }

        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);

        while (!queue.isEmpty()) {
            int levelSize = queue.size();
            List<Integer> currentLevel = new ArrayList<>();
            for (int i = 0; i < levelSize; i++) {
                TreeNode currentNode = queue.poll();
                currentLevel.add(currentNode.val);

                if (currentNode.left != null) {
                    queue.add(currentNode.left);
                }
                if (currentNode.right != null) {
                    queue.add(currentNode.right);
                }
            }
            result.add(currentLevel);
        }
        return result;
    }
}

public class FindDuplicates {
    public static List<Integer> findDuplicates(int[] nums) {
        List<Integer> duplicates = new ArrayList<>();
        if (nums == null || nums.length == 0) {
            return duplicates;
        }

        for (int i = 0; i < nums.length; i++) {
            int index = Math.abs(nums[i]) - 1;
            if (nums[index] < 0) {
                duplicates.add(Math.abs(nums[i]));
            } else {
                nums[index] = -nums[index];
            }
        }
        return duplicates;
    }

    public static void main(String[] args) {
        int[] nums = {4, 3, 2, 7, 8, 2, 3, 1};
        System.out.println("Duplicates in the array: " + findDuplicates(nums)); // Output: [2, 3]
    }

    /**
     * Checks if two strings are anagrams using Java 8 Streams.
     * It handles case and spaces by normalizing the string before sorting.
     *
     * @param s1 The first string.
     * @param s2 The second string.
     * @return true if they are anagrams, false otherwise.
     */
    public static boolean areAnagrams(String s1, String s2) {
        if (s1 == null || s2 == null) {
            return false;
        }

        // Helper function (Method Reference) to normalize and create the sorted key
        // 1. Convert to lowercase
        // 2. Remove all whitespace
        // 3. Convert to a character stream (IntStream)
        // 4. Sort the characters
        // 5. Collect the sorted characters back into a single String (the canonical key)
         Function<String, String> createCanonicalKey = (String s) -> s.toLowerCase()
                .replaceAll("\\s", "") // Remove all whitespace
                .chars()               // IntStream of character codes
                .sorted()              // Sorts the character codes (A < B < C...)
                .collect(StringBuilder::new, // Collector starts a new StringBuilder
                         StringBuilder::appendCodePoint, // Accumulator appends the character code
                         StringBuilder::append) // Combiner for parallel streams
                .toString();

        // Generate the canonical key for both strings
        String key1 = createCanonicalKey.apply(s1);
        String key2 = createCanonicalKey.apply(s2);

        // Anagrams will have identical canonical keys
        return key1.equals(key2);
    }

        // Test with your input
        boolean result1 = areAnagrams(s1, s2);

        // Test with a non-anagram pair
        boolean result2 = areAnagrams("hello", "world");

The most efficient way to solve this is using a HashMap. Iterate through the array once. For each number, calculate its complement (the value needed to reach the target). import java.util.HashMap;
import java.util.Map;

public class TwoSum {
    public int[] findTwoSum(int[] nums, int target) {
        Map<Integer, Integer> numMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            int complement = target - nums[i];
            if (numMap.containsKey(complement)) {
                return new int[] { numMap.get(complement), i };
            }
            numMap.put(nums[i], i);
        }
        throw new IllegalArgumentException("No two sum solution");
    }
}

        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                current.next = list1;
                list1 = list1.next;
            } else {
                current.next = list2;
                list2 = list2.next;
            }
            current = current.next;
        }

        // Append remaining nodes from either list

            current.next = list1;
        } else if (list2 != null) {
            current.next = list2;
        }

        return dummy.next;
    }

    public static int binarySearch(int[] arr, int target) {
        int low = 0;
        int high = arr.length - 1;

        while (low <= high) {
            int mid = low + (high - low) / 2;

            if (arr[mid] == target) {
                return mid; // Target found, return its index
            } else if (arr[mid] < target) {
                low = mid + 1; // Target is in the right half
            } else {
                high = mid - 1; // Target is in the left half
            }
        }
        return -1; // Target not found
    }

public class removeduplicatesinarrayinplace {

	   public static int removeDuplicates(int[] arr) {
	        if (arr == null || arr.length == 0) {
	            return 0; // No elements, so no duplicates. New length is 0.
	        }

	        // Step 1: Sort the array (if not already sorted)
	        // This is crucial for the two-pointer approach below.
	        Arrays.sort(arr);
	        System.out.println("Array after sorting: " + Arrays.toString(arr));

	        // Step 2: Two-pointer approach
	        int i = 0; // Pointer for the last unique element found (or next position for a unique element)

	        // Iterate with 'j' starting from the second element
	        for (int j = 1; j < arr.length; j++) {
	            // If the current element (arr[j]) is different from the last unique element (arr[i])
	            if (arr[j] != arr[i]) {
	                i++; // Move 'i' to the next position
	                arr[i] = arr[j]; // Place the new unique element at arr[i]
	            }
	            // If arr[j] == arr[i], it's a duplicate. We simply increment j and skip it.
	        }

	        // The number of unique elements is (i + 1)
	        return i + 1;
	    }

	public static String recur(String s) {
		// Base case: if the string has 0 or 1 character, it's already reversed.
		if (s.length() <= 1) {
			return s;
		}
		
		// Recursive step:
		// 1. Take the last character of the string.
		// 2. Recursively reverse the rest of the string (all but the last character).
		// 3. Concatenate them.
		
		return s.charAt(s.length() - 1) + recur(s.substring(0, s.length() - 1));
	}


















public class reversearrayinplace {
	  public static void reverse(int[] arr) {
	        // Handle null or empty arrays
	        if (arr == null || arr.length <= 1) {
	            return; // No need to reverse if array is null or has 0/1 element
	        }

	        int start = 0;
	        int end = arr.length - 1;

	        // Continue swapping until the pointers meet or cross
	        while (start < end) {
	            // Swap elements at start and end
	            int temp = arr[start]; // Store the value at 'start'
	            arr[start] = arr[end]; // Overwrite 'start' with 'end' value
	            arr[end] = temp;       // Overwrite 'end' with the stored 'start' value

	            // Move pointers towards the center
	            start++;
	            end--;
	        }
	    }

	    /**
	     * Checks if a given number is prime.
	     * This method implements a common optimization: checking divisibility only up to the square root of the number.
	     *
	     * @param number The number to check for primality.
	     * @return true if the number is prime, false otherwise.
	     */
	    public static boolean isPrime(int number) {
	        if (number <= 1) { // Numbers less than or equal to 1 are not prime
	            return false;
	        }
	        // Check for divisibility from 2 up to the square root of the number
	        // If a number has a divisor greater than its square root, it must also have a divisor
	        // smaller than its square root.
	        for (int i = 2; i * i <= number; i++) {
	            if (number % i == 0) { // If the number is divisible by i, it's not prime
	                return false;
	            }
	        }
	        return true; // If no divisors are found, the number is prime
	    }

```java
import java.util.Arrays;
import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

public class OrderStatisticsDirectMain {

    public static class Order {
        private String orderId;
        private double amount;

        public Order(String orderId, double amount) {
            this.orderId = orderId;
            this.amount = amount;
        }

        public String getOrderId() { return orderId; }
        public double getAmount() { return amount; }

        @Override
        public String toString() { return "Order{orderId='" + orderId + "', amount=" + amount + '}'; }
    }

    public static void main(String[] args) {
        System.out.println("--- Question 1: Summing and Averaging Order Amounts ---");

        List<Order> orders = Arrays.asList(
            new Order("A101", 150.75),
            new Order("A102", 200.50),
            new Order("A103", 50.25),
            new Order("A104", 100.00),
            new Order("A105", 200.50)
        );

        // Calculate total sum of amounts using reduce
        double totalSum = orders.stream()
                                .mapToDouble(Order::getAmount) // Use mapToDouble for primitive double stream
                                .sum(); // sum() is available on DoubleStream, IntStream, LongStream

        System.out.println("Total Order Amount (using mapToDouble.sum()): " + String.format("%.2f", totalSum));
        // Expected: 702.00

        // Alternative for sum using reducing (more generic)
        Optional<Double> totalSumReducing = orders.stream()
                                                  .map(Order::getAmount)
                                                  .reduce(Double::sum); // BinaryOperator for sum
        System.out.println("Total Order Amount (using reduce): " + String.format("%.2f", totalSumReducing.orElse(0.0)));


        // Calculate average amount using Collectors.averagingDouble
        Double averageAmount = orders.stream()
                                     .collect(Collectors.averagingDouble(Order::getAmount));

        System.out.println("Average Order Amount: " + String.format("%.2f", averageAmount));
        // Expected: 140.40 (702.00 / 5)

        // Using summary statistics (provides sum, average, min, max, count in one go)
        System.out.println("\n--- Using DoubleSummaryStatistics ---");
        Collectors.DoubleSummaryStatistics stats = orders.stream()
                                                        .collect(Collectors.summarizingDouble(Order::getAmount));
        System.out.println("Total Sum: " + String.format("%.2f", stats.getSum()));
        System.out.println("Average: " + String.format("%.2f", stats.getAverage()));
        System.out.println("Count: " + stats.getCount());
        System.out.println("Min: " + stats.getMin());
        System.out.println("Max: " + stats.getMax());

        System.out.println("\n");
    }
}
```

**Explanation:**

  * **`mapToDouble(Order::getAmount).sum()`**: For calculating sums, `mapToDouble()` creates a `DoubleStream` (a specialized stream for primitive `double` values). `DoubleStream` conveniently has a `sum()` terminal operation. This is generally more efficient for sums than `reduce(Double::sum)` on a `Stream<Double>` because it avoids auto-boxing/unboxing.
  * **`reduce(Double::sum)`**: A more general-purpose terminal operation that combines elements of a stream into a single result. `Double::sum` is a method reference to the `static` `sum` method in the `Double` class, which acts as a `BinaryOperator` for summing. It returns an `Optional` because the stream might be empty.
  * **`Collectors.averagingDouble(Order::getAmount)`**: A `Collector` specifically designed to compute the arithmetic mean (average) of a `double`-valued function applied to input elements. It handles the division and returns `Double` (which will be `0.0` for an empty stream).
  * **`Collectors.summarizingDouble(Order::getAmount)`**: This is highly efficient if you need multiple statistics (count, sum, min, max, average). It returns a `DoubleSummaryStatistics` object that holds all these values, performing only one pass over the data.

            this.name = name;

```java
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

public class ListToMapDirectMain {

    public static class Book {
        private int id;
        private String title;
        private String author;

        public Book(int id, String title, String author) {
            this.id = id;
            this.title = title;
            this.author = author;
        }

        public int getId() { return id; }
        public String getTitle() { return title; }
        public String getAuthor() { return author; }

        @Override
        public String toString() { return "Book{id=" + id + ", title='" + title + "', author='" + author + "'}"; }
    }

    public static void main(String[] args) {
        System.out.println("--- Question 4: Mapping to a Map with Collectors.toMap() ---");

        List<Book> books = Arrays.asList(
            new Book(1, "The Great Gatsby", "F. Scott Fitzgerald"),
            new Book(2, "1984", "George Orwell"),
            new Book(3, "To Kill a Mockingbird", "Harper Lee"),
            new Book(2, "Animal Farm", "George Orwell") // Duplicate ID for demonstration
        );

        Map<Integer, String> bookIdToTitleMap = books.stream()
            .collect(Collectors.toMap(
                Book::getId,         // Key Mapper: Function to extract the key (Book ID)
                Book::getTitle,      // Value Mapper: Function to extract the value (Book Title)
                (existing, replacement) -> existing // Merge Function: How to handle duplicate keys. Here, keep existing.
            ));

        System.out.println("Book ID to Title Map: " + bookIdToTitleMap);
        // Expected Output (order may vary): {1=The Great Gatsby, 2=1984, 3=To Kill a Mockingbird}
        // Note: "Animal Farm" (ID 2) is ignored due to the merge function.

        System.out.println("\n--- Handling Duplicates with 'replacement' (latest) ---");
        Map<Integer, String> bookIdToTitleMapKeepReplacement = books.stream()
            .collect(Collectors.toMap(
                Book::getId,
                Book::getTitle,
                (existing, replacement) -> replacement // Keep the replacement value
            ));
        System.out.println("Book ID to Title Map (keeping replacement): " + bookIdToTitleMapKeepReplacement);
        // Expected Output (order may vary): {1=The Great Gatsby, 2=Animal Farm, 3=To Kill a Mockingbird}


        // If there are no duplicate keys, you can omit the merge function:
        System.out.println("\n--- Example without duplicates ---");
        List<Book> uniqueBooks = Arrays.asList(
            new Book(10, "The Lord of the Rings", "J.R.R. Tolkien"),
            new Book(11, "Pride and Prejudice", "Jane Austen")
        );
        Map<Integer, String> uniqueBookMap = uniqueBooks.stream()
            .collect(Collectors.toMap(Book::getId, Book::getTitle));
        System.out.println("Unique Book Map: " + uniqueBookMap);

        System.out.println("\n");
    }
}
```

**Explanation:**

  * **`Collectors.toMap(keyMapper, valueMapper, mergeFunction)`**: This `Collector` is used to accumulate elements into a `Map`.
      * **`keyMapper` (`Book::getId`)**: A `Function` that extracts the key from each element.
      * **`valueMapper` (`Book::getTitle`)**: A `Function` that extracts the value from each element.
      * **`mergeFunction` (`(existing, replacement) -> existing`)**: This is crucial for handling duplicate keys. If two elements map to the same key, this `BinaryOperator` determines which value to keep. Here, `(existing, replacement) -> existing` means "if a duplicate key is found, keep the value that was already in the map (the existing one), discarding the new one." If you wanted to keep the *new* one, you'd use `(existing, replacement) -> replacement`. **If you don't provide a merge function and duplicate keys exist, `IllegalStateException` will be thrown.**

import java.util.List;
import java.util.StringJoiner;
import java.util.concurrent.atomic.AtomicInteger;

public class CustomIndexedJoinerDirectMain {

    public static void main(String[] args) {
        System.out.println("--- Question 1: Custom Collector to Join Strings with Index ---");

        List<String> fruits = Arrays.asList("Apple", "Banana", "Cherry", "Date");
        
        System.out.println("Indexed Fruits (Method 1): " + indexedJoinedString1);List<String> singleFruit = Arrays.asList("Mango");
        List<String> emptyFruits = Arrays.asList();

        // Method 1: Using IntStream and map (most common and readable for ordered lists)
        String indexedJoinedString1 = IntStream.range(0, fruits.size())
                                              .mapToObj(i -> (i + 1) + ". " + fruits.get(i))
                                              .collect(Collectors.joining(", "));

        String indexedJoinedStringSingle = IntStream.range(0, singleFruit.size())

        String indexedJoinedStringEmpty = IntStream.range(0, emptyFruits.size())
                                                   .mapToObj(i -> (i + 1) + ". " + emptyFruits.get(i))
                                                   .collect(Collectors.joining(", "));
        System.out.println("Indexed Empty Fruits: " + indexedJoinedStringEmpty);

        AtomicInteger counter = new AtomicInteger(0);

    }

.mapToObj(i -> (i + 1) + ". " + fruits.get(i)): For each index i, it creates a new string "(i+1). [fruitName]". This effectively transforms the IntStream into a Stream<String>.

.collect(Collectors.joining(", ")): Collects these newly formatted strings into a single string, using ", " as the delimiter.

Method 2 (Alternative using AtomicInteger with forEach): While not a pure Collector in the Stream.collect() sense, it's a common pattern to handle stateful operations during stream processing or with forEach.

          - Item: Laptop, Total Quantity: 3
          - Item: Charger, Total Quantity: 4
        */

        System.out.println("\n");
    }
}
Explanation:

Nested groupingBy Collectors: This is the key to multi-level grouping.

The outer groupingBy (Collectors.groupingBy(OrderLineItem::getProductId, ...)): This groups the initial stream of OrderLineItem objects by their productId. The result of this first step is implicitly a Map<String, List<OrderLineItem>> (where the list contains all items for a given productId).

The inner groupingBy (Collectors.groupingBy(OrderLineItem::getItemName, ...)): This is the downstream collector for the outer groupingBy. For each List<OrderLineItem> generated by the outer grouping, this inner collector is applied. It takes that list and further groups its elements by itemName. The result of this step for each outer group is a Map<String, List<OrderLineItem>> (where the inner list contains items for a specific productId AND itemName).

The innermost summingInt(OrderLineItem::getQuantity): This is the downstream collector for the inner groupingBy. For each List<OrderLineItem> generated by the inner grouping, it applies OrderLineItem::getQuantity to each item and sums up the quantities. This final result is an int (which Java auto-boxes to Integer for the map value).

The final return type Map<String, Map<String, Integer>> perfectly reflects this two-level grouping and aggregation.

**Question 2: Checking for Duplicate Elements in a List**

Given a `List<String>`, determine if it contains any duplicate elements. Return `true` if duplicates exist, `false` otherwise.

**Answer 2 (Direct in main):**

```javah
import java.util.Arrays;
import java.util.HashSet;
import java.util.List;
import java.util.Set;
import java.util.stream.Collectors; // Not strictly needed for the efficient solution, but common import

public class CheckDuplicatesDirectMain {

    public static void main(String[] args) {
        System.out.println("--- Question 2: Checking for Duplicate Elements in a List ---");

        List<String> listWithDuplicates = Arrays.asList("apple", "banana", "apple", "cherry", "banana");
        List<String> listWithoutDuplicates = Arrays.asList("apple", "banana", "cherry", "date");
        List<String> emptyList = Arrays.asList();
        List<String> singleElementList = Arrays.asList("single");

        // Efficient approach using Set and `anyMatch`
        boolean hasDuplicates1 = checkDuplicatesEfficient(listWithDuplicates);
        System.out.println("'" + listWithDuplicates + "' has duplicates: " + hasDuplicates1); // Expected: true

        boolean hasDuplicates2 = checkDuplicatesEfficient(listWithoutDuplicates);
        System.out.println("'" + listWithoutDuplicates + "' has duplicates: " + hasDuplicates2); // Expected: false

        boolean hasDuplicates3 = checkDuplicatesEfficient(emptyList);
        System.out.println("'" + emptyList + "' has duplicates: " + hasDuplicates3); // Expected: false

        boolean hasDuplicates4 = checkDuplicatesEfficient(singleElementList);
        System.out.println("'" + singleElementList + "' has duplicates: " + hasDuplicates4); // Expected: false


        // Alternative: Compare list size to distinct count (less efficient for large lists)
        System.out.println("\n--- Alternative: Comparing list size to distinct count ---");
        boolean hasDuplicatesAlt1 = listWithDuplicates.size() != listWithDuplicates.stream().distinct().count();
        System.out.println("'" + listWithDuplicates + "' has duplicates (Alt): " + hasDuplicatesAlt1); // Expected: true

        boolean hasDuplicatesAlt2 = listWithoutDuplicates.size() != listWithoutDuplicates.stream().distinct().count();
        System.out.println("'" + listWithoutDuplicates + "' has duplicates (Alt): " + hasDuplicatesAlt2); // Expected: false

        System.out.println("\n");
    }

    // Helper method for the efficient duplicate check
    private static boolean checkDuplicatesEfficient(List<String> list) {
        Set<String> seen = new HashSet<>();
        // anyMatch returns true as soon as a duplicate is found, short-circuiting the stream
        return list.stream().anyMatch(e -> !seen.add(e));
    }
}
```

**Explanation:**

  * **Efficient Approach (`anyMatch` with `HashSet`):** This is generally the most efficient stream-based way to detect duplicates because of short-circuiting.
      * `Set<String> seen = new HashSet<>();`: A `HashSet` is used to keep track of elements encountered so far. `Set.add(element)` returns `true` if the element was successfully added (i.e., it was not already present) and `false` if the element was already in the set (i.e., it's a duplicate).
      * `list.stream().anyMatch(e -> !seen.add(e))`:
          * `anyMatch(Predicate)`: A terminal short-circuiting operation that returns `true` if *any* element matches the given `Predicate`. It stops processing as soon as a `true` is found.
          * `e -> !seen.add(e)`: The `Predicate`. For each element `e`:
              * `seen.add(e)`: Attempts to add `e` to the `seen` set.
              * `!seen.add(e)`: If `add()` returns `false` (meaning `e` was already in `seen`), then `!false` is `true`. This `true` is what `anyMatch` looks for. So, `anyMatch` will return `true` immediately when the first duplicate is encountered.
  * **Alternative (`distinct().count()`):**
      * `list.stream().distinct().count()`: Counts the number of unique elements.
      * `list.size() != uniqueCount`: If the original list's size is different from the count of unique elements, then duplicates must exist.
      * **Performance Note:** This approach processes the *entire* list to create the distinct set and count it, even if a duplicate is found very early. For very large lists with early duplicates, the `anyMatch` approach is significantly faster due to short-circuiting.

**Question 5: Performing a Side-Effect Operation with `peek` (Debugging/Logging Example)**

You have a `List<String>`. Filter out strings shorter than 5 characters, convert the remaining to uppercase, and then print each intermediate string before final collection using `peek`.

**Answer 5 (Direct in main):**

```java
import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class PeekSideEffectDirectMain {

    public static void main(String[] args) {
        System.out.println("--- Question 5: Performing a Side-Effect Operation with 'peek' ---");

        List<String> words = Arrays.asList("apple", "banana", "cat", "dog", "elephant", "fish", "grape");

        System.out.println("Original words: " + words);
        System.out.println("Processing with peek:");

        List<String> processedWords = words.stream()
            .peek(s -> System.out.println("Initial: " + s)) // Peek at original string
            .filter(s -> s.length() >= 5) // Filter: keep only words with length >= 5
            .peek(s -> System.out.println("After filter (length >= 5): " + s)) // Peek after filter
            .map(String::toUpperCase) // Transform: convert to uppercase
            .peek(s -> System.out.println("After map (uppercase): " + s)) // Peek after map
            .collect(Collectors.toList());

        System.out.println("\nFinal processed words: " + processedWords);
        // Expected output for processedWords: [APPLE, BANANA, ELEPHANT, GRAPE]

        /*
        Example Console Output for peek (interleaved due to stream execution):
        Initial: apple
        After filter (length >= 5): apple

        Initial: banana
        After filter (length >= 5): banana
        After map (uppercase): BANANA
        Initial: cat
        Initial: dog
        Initial: elephant
        After filter (length >= 5): elephant
        After map (uppercase): ELEPHANT
        Initial: fish
        Initial: grape
        After filter (length >= 5): grape
        After map (uppercase): GRAPE

        Final processed words: [APPLE, BANANA, ELEPHANT, GRAPE]

    }

```

**Explanation:**

  * **`peek(Consumer action)`**: This is an **intermediate operation** that allows you to perform an action on each element as it passes through the stream pipeline, *without modifying the elements themselves*.

      * It's named "peek" because you're just taking a quick look at the elements as they go by.
  * **`filter(s -> s.length() >= 5)`**: Removes "cat", "dog", "fish".
  * **`map(String::toUpperCase)`**: Converts "apple" to "APPLE", "banana" to "BANANA", etc.
  * By strategically placing `peek` operations, you can visualize the state of elements at different stages of the stream pipeline, which is incredibly useful for understanding complex stream logic or debugging unexpected results.
  * **Important Note:** `peek` is for side-effects and debugging. It should *not* be used to modify the elements of the stream or perform the core logic. For actual transformation, use `map`; for filtering, use `filter`.

public class OneTransaction {
    public static int maxProfit(int[] prices) {
        int minPrice = Integer.MAX_VALUE, profit = 0;
        for (int price : prices) {
            minPrice = Math.min(minPrice, price);
            profit = Math.max(profit, price - minPrice);
        }
        return profit;
    }

    public static void main(String[] args) {
        int[] prices = {8, 3, 5, 1, 6};
        System.out.println("Max Profit = " + maxProfit(prices)); // Output -> 5
    }

public class KTransactions {
    public static int maxProfit(int k, int[] prices) {
        int n = prices.length;
        if (n == 0 || k == 0) return 0;

        // If k is large, it’s same as unlimited
        if (k >= n / 2) {
            int profit = 0;
            for (int i = 1; i < n; i++) {
                if (prices[i] > prices[i - 1]) {
                    profit += prices[i] - prices[i - 1];
                }
            }
            return profit;
        }
 // dp[t][d] represents the maximum profit using at most 't' transactions
        // up to day 'd'.
        int[][] dp = new int[k + 1][n];

        // Fill the DP table
        for (int t = 1; t <= k; t++) {
            // maxDiff tracks the maximum profit from t-1 transactions minus
            // the cost of buying on a previous day.
            int maxDiff = -prices[0];
            
            for (int d = 1; d < n; d++) {
                // Option 1: No transaction on day 'd'. Profit is the same as the previous day.
                // Option 2: Sell on day 'd'. This means we bought on a previous day 'j'
                // and made t-1 transactions before day 'j'. The profit would be
                // prices[d] + (max profit from t-1 transactions before day d - prices[j]).
                // maxDiff helps us find the best previous 'buy' day efficiently.
                dp[t][d] = Math.max(dp[t][d - 1], prices[d] + maxDiff);

                // Update maxDiff for the next day. This considers two possibilities
                // for the next 'buy' opportunity:
                // 1. We completed t-1 transactions by day 'd' and now buy on day 'd'.
                //    The total profit so far would be dp[t-1][d] - prices[d].
                // 2. The previous maxDiff from a day before 'd' is still the best.
                maxDiff = Math.max(maxDiff, dp[t - 1][d] - prices[d]);
            }
        }
        
        // The final result is the maximum profit with at most 'k' transactions
        // by the last day.
        return dp[k][n - 1];
    }    }

    public static void main(String[] args) {
        int[] prices = {1, 10, 2, 15};
        int k = 2;
        System.out.println("Max Profit = " + maxProfit(k, prices)); // Output -> 22
    }
}

public class WithFee {

        int hold = -prices[0], cash = 0;
        for (int i = 1; i < prices.length; i++) {
            cash = Math.max(cash, hold + prices[i] - fee);
            hold = Math.max(hold, cash - prices[i]);
        }
        return cash;
    }

public class WithCooldown {
    public static int maxProfit(int[] prices) {
        if (prices.length == 0) return 0;
        int n = prices.length;
        int[] hold = new int[n];
        int[] sold = new int[n];
        int[] rest = new int[n];

        hold[0] = -prices[0];
        for (int i = 1; i < n; i++) {
            hold[i] = Math.max(hold[i - 1], rest[i - 1] - prices[i]);
            sold[i] = hold[i - 1] + prices[i];
            rest[i] = Math.max(rest[i - 1], sold[i - 1]);
        }
        return Math.max(sold[n - 1], rest[n - 1]);
    }

    public static void main(String[] args) {
        int[] prices = {1, 2, 3, 0, 2};
        System.out.println("Max Profit = " + maxProfit(prices)); // Output -> 3
    }
}
