Using Guava Cache (Recommended)
Guava's CacheBuilder is designed for this exact scenario, providing automatic, thread-safe, time-based eviction without manual scheduling.

Java

import com.google.common.cache.Cache;
import com.google.common.cache.CacheBuilder;
import java.util.concurrent.TimeUnit;

// A simple structure to hold the report data
public class ReportData { /* fields like String content, long size, etc. */ }

public class AutoClearingReportCache {

    // 1. Define the Cache: Key is the application ID (String), Value is the report data
    private final Cache<String, ReportData> temporaryCache = CacheBuilder.newBuilder()
            // Clears entries 5 minutes after they were written/inserted
            .expireAfterWrite(5, TimeUnit.MINUTES)
            // Optional: Limit the total size of the cache by number of entries
            .maximumSize(5000) 
            .build();

    /**
     * Stores a report associated with a loan application ID.
     * @param applicationId The ID of the application.
     * @param reportData The large report data object.
     */
    public void putReport(String applicationId, ReportData reportData) {
        temporaryCache.put(applicationId, reportData);
    }
